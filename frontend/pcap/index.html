// prerequisites
Linux sudo apt-get install -y libpcap-dev // must run with sudo
macOS brew install libpcap // must run with sudo
Windows Download and install Npcap from npcap.com (same as Wireshark) // must run as Administrator
// installation
// option 1 — download binary (recommended)
PlatformFile
Linux (amd64) pcap-agent-linux-amd64
Windows (amd64) pcap-agent-windows-amd64.exe
macOS (Intel) pcap-agent-darwin-amd64
macOS (Apple Silicon) pcap-agent-darwin-arm64
// option 2 — build from source
git clone https://github.com/Ahlyx/pcap-agent.git cd pcap-agent go build -o pcap-agent ./cmd/agent
// usage
sudo ./pcap-agent list-interfaces sudo ./pcap-agent start sudo ./pcap-agent start --interface eth0 sudo ./pcap-agent start --port 8888 sudo ./pcap-agent start --relay
